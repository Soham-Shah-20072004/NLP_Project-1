\section*{4. Cosine Similarity and Ranking}
\textbf{Query Representation (TF-IDF):}
\begin{itemize}
    \item TF-IDF("star") = $1 \times 0.176 = 0.176$
    \item TF-IDF("light") = $1 \times 0.477 = 0.477$
\end{itemize}
Query Vector $Q$ Norm: $||Q|| = \sqrt{0.176^2 + 0.477^2} = \sqrt{0.031 + 0.228} \approx 0.509$

\vspace{1em}
\textbf{Document Norms:}
\begin{itemize}
    \item $||d_1|| = \sqrt{0.176^2 + 5(0.477^2)} = \sqrt{0.031 + 1.138} \approx 1.081$
    \item $||d_2|| = \sqrt{0.176^2 + 6(0.477^2)} = \sqrt{0.031 + 1.365} \approx 1.181$
    \item $||d_3|| = \sqrt{7(0.477^2)} \approx 1.262$
\end{itemize}

\vspace{1em}
\textbf{Cosine Similarity Calculations:}
\begin{align*}
\cos(d_1, Q) &= \frac{(0.176 \times 0.176) + (0.477 \times 0.477)}{1.081 \times 0.509} = \frac{0.259}{0.550} \approx \mathbf{0.471} \\
\cos(d_2, Q) &= \frac{(0.176 \times 0.176) + (0)}{1.181 \times 0.509} = \frac{0.031}{0.601} \approx \mathbf{0.052} \\
\cos(d_3, Q) &= \frac{0}{1.262 \times 0.509} = \mathbf{0}
\end{align*}

\textbf{Final Ranking:} $\mathbf{d_1 > d_2 > d_3}$ \\
\textbf{Is the ranking desirable?} Yes. The query "star light" heavily implies an astronomical context. Document $d_1$ correctly ranks first as it discusses the sun providing heat and light, whereas $d_2$ is about a Hollywood actor. 

\section*{5. Analysis of Word Sense Ambiguity}
\textbf{Query:} "movie star"
\begin{itemize}
    \item \textbf{Ideally retrieved:} $d_2$ should be retrieved, as it contextually aligns with the entertainment industry. 
    \item \textbf{System retrieval:} The system will successfully retrieve $d_2$ because "movie" and "star" both appear in $d_2$'s vector.
    \item \textbf{Effect of Ambiguity:} The word "star" is polysemous (it has multiple meanings: a celestial body vs. a celebrity). If the query were shorter, such as just "star", the VSM would match both $d_1$ and $d_2$ equally purely based on token overlap, despite them belonging to entirely different semantic domains. The VSM strictly relies on lexical matching and lacks the contextual awareness to disambiguate which sense of "star" the user intended.
\end{itemize}
